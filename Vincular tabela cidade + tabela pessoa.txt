CREATE TABLE CIDADE (

	ID SERIAL,
	DESCRICAO VARCHAR (150) NOT NULL,
	CONSTRAINT PK_CIDADE PRIMARY KEY (ID)

);

CREATE TABLE PESSOA (

	ID SERIAL,
	NOME VARCHAR (50),
	IDADE INTEGER,
	CIDADE_ID INTEGER,
	
	CONSTRAINT PK_PESSOA PRIMARY KEY (ID),
	CONSTRAINT FK_PESSOA_CIDADE FOREIGN KEY (CIDADE_ID) REFERENCES CIDADE (ID)
	
);
